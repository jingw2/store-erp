# Store ERP — Plan 1: Foundation (Monorepo, Database, Auth, Multi-tenancy)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Bootstrap the Turborepo monorepo, define the complete Prisma schema, and implement JWT authentication with multi-tenant row-level isolation — producing a runnable API that enforces tenant boundaries and role-based access.

**Architecture:** Turborepo with `apps/api` (NestJS + Fastify) and `packages/types`. Prisma ORM on PostgreSQL with a middleware layer that auto-injects `tenantId` on every query (SUPER_ADMIN bypasses this). JWT auth issues tokens containing `userId`, `tenantId`, and `globalRole`; NestJS Guards enforce roles and tenant context on every route.

**Tech Stack:** pnpm workspaces, Turborepo, NestJS 10, Fastify, Prisma 5, PostgreSQL 16, `@nestjs/jwt`, `@nestjs/passport`, `passport-jwt`, Jest, Supertest, Docker Compose.

**Plan series:** This is Plan 1 of 7.
- Plan 1 (this): Foundation — monorepo, schema, auth, multi-tenancy
- Plan 2: Master Data API — tenants, stores, users, products, materials, recipes
- Plan 3: Inventory Management — stock CRUD, transactions, CSV import
- Plan 4: POS Integration & Sales — webhook adapters, auto-deduction
- Plan 5: Stocktake — session state machine, snapshot, approval
- Plan 6: Restocking & Notifications — BullMQ jobs, WhatsApp, Email
- Plan 7: Frontend Shell — Next.js, i18n, sidebar, UI component library

// ─── Tenant & User ────────────────────────────────────────────────────────────

model Tenant {
  id            String   @id @default(cuid())
  name          String
  slug          String   @unique
  defaultLocale String   @default("zh")
  plan          String   @default("starter")
  createdAt     DateTime @default(now())

  stores         Store[]
  users          User[]
  regions        TenantRegion[]
  products       Product[]
  materials      Material[]
  productRecipes ProductRecipe[]
  storeInventory StoreInventory[]
  transactions   InventoryTransaction[]
  saleEvents     SaleEvent[]
  stocktakes     StocktakeSession[]
  restockSuggestions RestockSuggestion[]
  webhookLogs    WebhookLog[]
}

model User {
  id           String     @id @default(cuid())
  tenantId     String?    // null for SUPER_ADMIN
  email        String     @unique
  passwordHash String
  name         String
  globalRole   GlobalRole @default(STAFF)
  locale       String     @default("zh")
  createdAt    DateTime   @default(now())

  tenant       Tenant?    @relation(fields: [tenantId], references: [id])
  storeUsers   StoreUser[]
  userRegions  UserRegion[]
  transactions InventoryTransaction[]
  stocktakeCreated  StocktakeSession[] @relation("CreatedBy")
  stocktakeConfirmed StocktakeSession[] @relation("ConfirmedBy")
}

model TenantRegion {
  id       String @id @default(cuid())
  tenantId String
  name     String // unique within tenant

  tenant      Tenant       @relation(fields: [tenantId], references: [id])
  stores      Store[]
  userRegions UserRegion[]

  @@unique([tenantId, name])
}

model UserRegion {
  id       String @id @default(cuid())
  userId   String
  tenantId String
  region   String // must match a TenantRegion.name for this tenant

  user         User         @relation(fields: [userId], references: [id])
  tenantRegion TenantRegion @relation(fields: [tenantId, region], references: [tenantId, name])

  @@unique([userId, region])
}

model Store {
  id        String   @id @default(cuid())
  tenantId  String
  name      String
  region    String   // references TenantRegion.name
  timezone  String   @default("Asia/Kuala_Lumpur")
  status    String   @default("active")
  createdAt DateTime @default(now())

  tenant          Tenant           @relation(fields: [tenantId], references: [id])
  storeUsers      StoreUser[]
  storeInventory  StoreInventory[]
  saleEvents      SaleEvent[]
  stocktakes      StocktakeSession[]
  restockSuggestions RestockSuggestion[]
  webhookLogs     WebhookLog[]
  transactions    InventoryTransaction[]
}

model StoreUser {
  id        String    @id @default(cuid())
  userId    String
  storeId   String
  tenantId  String
  role      StoreRole

  user  User  @relation(fields: [userId], references: [id])
  store Store @relation(fields: [storeId], references: [id])

  @@unique([userId, storeId])
}

  id       String  @id @default(cuid())
  tenantId String
  name     String
  nameEn   String
  sku      String
  category String  @default("")

  name     String
  nameEn   String

  storeId     String

- [ ] **Step 3: Create `apps/api/src/prisma/prisma.service.ts`**

> **Note on concurrency safety:** `tenantContext` is stored via `AsyncLocalStorage` so each concurrent request gets its own isolated context. A plain instance property on a singleton would cause tenant context to leak across concurrent requests.

```typescript
import { Injectable, OnModuleInit, OnModuleDestroy } from '@nestjs/common';
import { PrismaClient } from '@prisma/client';
import { AsyncLocalStorage } from 'async_hooks';
import { GlobalRole, isSuperAdmin } from '@store-erp/types';

interface TenantContext {
  tenantId: string | null;
  isSuperAdmin: boolean;
}

const tenantStorage = new AsyncLocalStorage<TenantContext>();

@Injectable()
export class PrismaService
  extends PrismaClient
  implements OnModuleInit, OnModuleDestroy
{
  constructor() {
    super();

    // Multi-tenant middleware: inject tenantId on every query
    // unless the caller is SUPER_ADMIN.
    // Context is stored in AsyncLocalStorage so concurrent requests
    // never share each other's tenant context.
    this.$use(async (params, next) => {
      const ctx = tenantStorage.getStore();

      if (!ctx || ctx.isSuperAdmin) {
        // SUPER_ADMIN carve-out: no tenantId injection
        return next(params);
      }

      const tenantId = ctx.tenantId;
      if (!tenantId) return next(params);

      // Inject on create / createMany
      if (params.action === 'create') {
        params.args.data = { ...params.args.data, tenantId };
      }
      if (params.action === 'createMany') {
        params.args.data = params.args.data.map((d: Record<string, unknown>) => ({
          ...d,
          tenantId,
        }));
      }

      // Inject on reads / updates / deletes
      const readActions = ['findFirst', 'findUnique', 'findMany', 'count', 'aggregate'];
      const writeActions = ['update', 'updateMany', 'delete', 'deleteMany', 'upsert'];

      if (readActions.includes(params.action) || writeActions.includes(params.action)) {
        params.args.where = { ...params.args.where, tenantId };
      }

      return next(params);
    });
  }

  async onModuleInit() {
    await this.$connect();
  }

  async onModuleDestroy() {
    await this.$disconnect();
  }

  /**
   * Set tenant context for the current async execution context (request).
   * Wraps the callback in AsyncLocalStorage so the context is isolated
   * per request and never leaks across concurrent requests.
   */
  runWithTenantContext<T>(
    tenantId: string | null,
    role: GlobalRole,
    fn: () => T,
  ): T {
    return tenantStorage.run({ tenantId, isSuperAdmin: isSuperAdmin(role) }, fn);
  }

  getTenantContext(): TenantContext | undefined {
    return tenantStorage.getStore();
  }
}
```

import { Injectable, UnauthorizedException } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import * as bcrypt from 'bcrypt';
import { PrismaService } from '../prisma/prisma.service';
import { AuthUser, JwtPayload } from '@store-erp/types';

@Injectable()
export class AuthService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly jwt: JwtService,
  ) {}

  async validateUser(email: string, password: string): Promise<AuthUser | null> {
    const user = await this.prisma.user.findUnique({ where: { email } });
    if (!user) return null;

    const valid = await bcrypt.compare(password, user.passwordHash);
    if (!valid) return null;

    return {
      userId: user.id,
      tenantId: user.tenantId,
      role: user.globalRole as AuthUser['role'],
    };
  }

  async login(user: AuthUser): Promise<{ accessToken: string }> {
    const payload: JwtPayload = {
      sub: user.userId,
      tenantId: user.tenantId,
      role: user.role,
    };
    return { accessToken: this.jwt.sign(payload) };
  }

  async hashPassword(password: string): Promise<string> {
    return bcrypt.hash(password, 10);
  }
}

```typescript
import { Injectable, ExecutionContext } from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';
import { PrismaService } from '../prisma/prisma.service';
import { AuthUser } from '@store-erp/types';

@Injectable()
export class JwtAuthGuard extends AuthGuard('jwt') {
  constructor(private readonly prisma: PrismaService) {
    super();
  }

  // canActivate wraps the entire request handler in runWithTenantContext
  // so the AsyncLocalStorage context is available throughout the call chain.
  async canActivate(context: ExecutionContext): Promise<boolean> {
    const canActivate = await super.canActivate(context);
    if (!canActivate) return false;

    const request = context.switchToHttp().getRequest<{ user: AuthUser; tenantContextSet?: boolean }>();
    const user = request.user;

    return new Promise((resolve, reject) => {
      this.prisma.runWithTenantContext(user.tenantId, user.role, () => {
        // Mark that context is set; actual handler runs inside this closure
        // by NestJS interceptors inheriting the async context.
        request.tenantContextSet = true;
        resolve(true);
      });
    });
  }
}
```

> **Implementation note:** Because Fastify/NestJS route handlers run in a continuation of the same async context as `canActivate`, `AsyncLocalStorage` context set here is available to all downstream services (including PrismaService middleware) for the lifetime of the request without any additional wiring.

import { Injectable, CanActivate, ExecutionContext } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { ROLES_KEY } from './roles.decorator';
import { GlobalRole, AuthUser } from '@store-erp/types';

@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const required = this.reflector.getAllAndOverride<GlobalRole[]>(ROLES_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);

    if (!required || required.length === 0) return true;

    const request = context.switchToHttp().getRequest<{ user: AuthUser }>();
    return required.includes(request.user.role);
  }
}
